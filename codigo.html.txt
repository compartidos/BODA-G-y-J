<script>
  // ⚠️ REEMPLAZÁ con la URL de tu deployment de Apps Script
  const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbzDztTJCn6wCWiSODtlaZ5u1G1l5so4Jt91_wK-KgV7UdjKIkRPuYjYaSeQNDLQzm9g/exec';

  let selectedFiles = [];

  // ── Drop zone ──
  const dropZone = document.getElementById('dropZone');
  const fileInput = document.getElementById('fileInput');
  dropZone.addEventListener('click', () => fileInput.click());
  fileInput.addEventListener('change', e => handleFiles(e.target.files));
  dropZone.addEventListener('dragover', e => { e.preventDefault(); dropZone.classList.add('drag-over'); });
  dropZone.addEventListener('dragleave', () => dropZone.classList.remove('drag-over'));
  dropZone.addEventListener('drop', e => {
    e.preventDefault(); dropZone.classList.remove('drag-over');
    handleFiles(e.dataTransfer.files);
  });

  function handleFiles(files) {
    selectedFiles = Array.from(files).filter(f => f.type.startsWith('image/'));
    const strip = document.getElementById('previewStrip');
    strip.innerHTML = '';
    selectedFiles.forEach(f => {
      const img = document.createElement('img');
      img.src = URL.createObjectURL(f);
      strip.appendChild(img);
    });
    document.getElementById('uploadBtn').disabled = selectedFiles.length === 0;
    document.getElementById('progressMsg').textContent =
      selectedFiles.length ? `${selectedFiles.length} foto(s) seleccionada(s)` : '';
  }

  // ── Subir todas ──
  async function uploadAll() {
    if (!selectedFiles.length) return;
    const btn = document.getElementById('uploadBtn');
    const msg = document.getElementById('progressMsg');
    btn.disabled = true;
    const total = selectedFiles.length;

    for (let i = 0; i < total; i++) {
      msg.textContent = `Subiendo ${i + 1} de ${total}...`;
      await uploadOne(selectedFiles[i]);
    }

    msg.textContent = '✓ ¡Fotos subidas con éxito!';
    document.getElementById('previewStrip').innerHTML = '';
    selectedFiles = [];
    setTimeout(loadGallery, 1500);
  }

  function uploadOne(file) {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onload = async function(e) {
        const base64 = e.target.result.split(',')[1];
        try {
          const res = await fetch(SCRIPT_URL, {
            method: 'POST',
            body: JSON.stringify({
              nombre: file.name,
              tipo: file.type,
              datos: base64
            })
          });
          const json = await res.json();
          json.ok ? resolve(json) : reject(json.error);
        } catch(err) { reject(err); }
      };
      reader.readAsDataURL(file);
    });
  }

  // ── Cargar galería ──
  async function loadGallery() {
    try {
      const res = await fetch(SCRIPT_URL + '?action=list');
      const json = await res.json();
      const gallery = document.getElementById('gallery');

      if (!json.ok || !json.fotos.length) {
        gallery.innerHTML = '<div class="empty-gallery">Aún no hay fotos. ¡Sé el primero en subir la tuya!</div>';
        return;
      }

      gallery.innerHTML = json.fotos.map(f => `
        <div class="masonry-item" onclick="openLightbox('${f.url}')">
          <img src="${f.url}" alt="${f.nombre}" loading="lazy">
        </div>
      `).join('');
    } catch(e) {
      document.getElementById('gallery').innerHTML = '<div class="empty-gallery">Error al cargar fotos.</div>';
    }
  }

  // ── Lightbox ──
  function openLightbox(url) {
    document.getElementById('lightboxImg').src = url;
    document.getElementById('lightbox').classList.add('open');
  }
  function closeLightbox() {
    document.getElementById('lightbox').classList.remove('open');
    document.getElementById('lightboxImg').src = '';
  }
  document.getElementById('lightbox').addEventListener('click', function(e) {
    if (e.target === this) closeLightbox();
  });

  // Cargar al iniciar
  loadGallery();
</script>
